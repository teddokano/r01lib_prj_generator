56void NAFE13388_Base::logical_ch_config( int ch, uint16_t cc0, uint16_t cc1, uint16_t cc2, uint16_t cc3 )
57{
58 constexpr double pga_gain[] = { 0.2, 0.4, 0.8, 1, 2, 4, 8, 16 };
59
60 write_r16( ch );
61
62 write_r16( 0x0020, cc0 );
63 write_r16( 0x0021, cc1 );
64 write_r16( 0x0022, cc2 );
65 write_r16( 0x0023, cc3 );
66
67 uint16_t mask = 1;
68 uint16_t bits = read_r16( 0x0024 ) | (mask << ch);
69 enabled_channels = 0;
70
71 for ( int i = 0; i < 16; i++ ) {
72 if ( bits & (mask << i) )
73 enabled_channels++;
74 }
75
76 write_r16( 0x0024, bits );
77
78 if ( cc0 & 0x0010 )
79 coeff_uV[ ch ] = ((10.0 / (double)(1L << 24)) / pga_gain[ (cc0 >> 5) & 0x7 ]) * 1e6;
80 else
81 coeff_uV[ ch ] = (4.0 / (double)(1L << 24)) * 1e6;
82}
