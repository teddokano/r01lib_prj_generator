120void NAFE13388_Base::logical_ch_config( int ch, const uint16_t (&cc)[ 4 ] )
121{
122 command( ch );
123
124 for ( auto i = 0; i < 4; i++ )
125 reg( CH_CONFIG0 + i, cc[ i ] );
126
127 const uint16_t setbit = 0x1 << ch;
128 const uint16_t bits = bit_op( CH_CONFIG4, ~setbit, setbit );
129
130 enabled_channels = bit_count( bits );
131
132 if ( cc[ 0 ] & 0x0010 )
133 coeff_uV[ ch ] = ((10.0 / (double)(1L << 24)) / pga_gain[ (cc[ 0 ] >> 5) & 0x7 ]) * 1e6;
134 else
135 coeff_uV[ ch ] = (4.0 / (double)(1L << 24)) * 1e6;
136
137 ch_delay[ ch ] = calc_delay( ch );
138}

140double NAFE13388_Base::calc_delay( int ch )
141{
142 constexpr static double data_rates[] = { 288000, 192000, 144000, 96000, 72000, 48000, 36000, 24000,
143 18000, 12000, 9000, 6000, 4500, 3000, 2250, 1125,
144 562.5, 400, 300, 200, 100, 60, 50, 30,
145 25, 20, 15, 10, 7.5, };
146 constexpr static uint16_t delays[] = { 0, 2, 4, 6, 8, 10, 12, 14,
147 16, 18, 20, 28, 38, 40, 42, 56,
148 64, 76, 90, 128, 154, 178, 204, 224,
149 256, 358, 512, 716,
150 1024, 1664, 3276, 7680, 19200, 23040, };
151
152 command( ch );
153
154 uint16_t ch_config1 = reg( CH_CONFIG1 );
155 uint16_t ch_config2 = reg( CH_CONFIG2 );
156
157 uint8_t adc_data_rate = (ch_config1 >> 3) & 0x001F;
158 uint8_t adc_sinc = (ch_config1 >> 0) & 0x0007;
159 uint8_t ch_delay = (ch_config2 >> 10) & 0x003F;
160 bool adc_normal_setting = (ch_config2 >> 9) & 0x0001;
161 bool ch_chop = (ch_config2 >> 7) & 0x0001;
162
163 double base_freq = data_rates[ adc_data_rate ];
164 double delay_setting = delays[ ch_delay ] / 4608000.00;
165
166 if ( (28 < adc_data_rate) || (4 < adc_sinc) || ((adc_data_rate < 12) && (adc_sinc)) )
167 return 0.00;
168
169 if ( !adc_normal_setting )
170 base_freq /= (adc_sinc + 1);
171
172 if ( ch_chop )
173 base_freq /= 2;
174
175#if 0
176 printf( "base_freq = %lf\r\n", base_freq );
177 printf( "delay_setting = %lf\r\n", delay_setting );
178 printf( "total delay = %lf\r\n", (1 / base_freq) + delay_setting );
179#endif
180
181 return (1 / base_freq) + delay_setting;
182}

259void NAFE13388_Base::gain_offset_coeff( const ref_points &ref )
260{
261 constexpr double pga1x_voltage = 5.0;
262 constexpr int adc_resolution = 24;
263 constexpr double pga_gain_setting = 0.2;
264
265 constexpr double fullscale_voltage = pga1x_voltage / pga_gain_setting;
266
267 double fullscale_data = pow( 2, (adc_resolution - 1) );
268 double ref_data_span = ref.high.data - ref.low.data;
269 double ref_voltage_span = ref.high.voltage - ref.low.voltage;
270
271 double dv_slope = ref_data_span / ref_voltage_span;
272 double custom_gain = dv_slope * (fullscale_voltage / fullscale_data);
273 double custom_offset = (dv_slope * ref.low.voltage - ref.low.data) / custom_gain;
274
275 int32_t gain_coeff_cal = reg( GAIN_COEFF0 + ref.cal_index );
276 int32_t offsset_coeff_cal = reg( OFFSET_COEFF0 + ref.cal_index );
277 int32_t gain_coeff_new = round( gain_coeff_cal * custom_gain );
278 int32_t offset_coeff_new = round( custom_offset - offsset_coeff_cal );
279
280#if 0
281 printf( "ref_point_high = %8ld @%6.3lf\r\n", ref.high.data, ref.high.voltage );
282 printf( "ref_point_low = %8ld @%6.3lf\r\n", ref.low.data, ref.low.voltage );
283 printf( "gain_coeff_new = %8ld\r\n", gain_coeff_new );
284 printf( "offset_coeff_new = %8ld\r\n", offset_coeff_new );
285#endif
286
287 reg( GAIN_COEFF0 + ref.coeff_index, gain_coeff_new );
288 reg( OFFSET_COEFF0 + ref.coeff_index, offset_coeff_new );
289}

291NAFE13388_Base::CalibrationError NAFE13388_Base::self_calibrate( GainPGA pga_gain_index, int channel_selection, int input_select, double reference_source_voltage, bool use_positive_side )
292{
293 constexpr auto low_gain_index = 2;
294 auto channel_in_use = false;
295 ch_setting_t tmp_ch_config;
296 int gain_index = static_cast<int>( pga_gain_index );
297
298 // logical channel selection to perform the self-calibration
299 // if the chennel in-use, save channel setting to temporal memory
300
301 if ( reg( CH_CONFIG4 ) & (0x1 << channel_selection) )
302 {
303 channel_in_use = true;
304
305 command( channel_selection );
306
307 for ( auto i = 0; i < 4; i++ )
308 tmp_ch_config[ i ] = reg( CH_CONFIG0 + i );
309 }
310
311 // if user doesn't specify the channel and voltage, use REFH or REFL
312
313 if ( !input_select )
314 {
315 bool low_gain = (gain_index <= low_gain_index);
316
317 input_select = low_gain ? 0x5 : 0x6;
318 reference_source_voltage = (reg( low_gain ? OPT_COEF1 : OPT_COEF2 ) * 5.00) / (double)(1 << 24);
319
320#if 1
321 printf( "==== self-calibration for PGA gain setting: x%3.1lf\r\n", pga_gain[ gain_index ] );
322 printf( "gain = %s\r\n", low_gain ? "low" : "high" );
323 printf( "REF[H|L] = %10.8lfV\r\n", reference_source_voltage );
324#endif
325 }
326
327 // logical channel settings
328 // Total 3 settings are prepared to measure reference_voltage, internal-GND and AICOM
329
330 const uint16_t REF_GND = 0x0011 | (gain_index << 5);
331 const uint16_t REF_V = (input_select << (use_positive_side ? 12 : 8)) | REF_GND;
332 const uint16_t REF_COM = 0x7700 | REF_GND;
333 const uint16_t ch_config1 = (gain_index << 12) | 0x00E4;
334 constexpr uint16_t ch_config2 = 0x8480;
335 constexpr uint16_t ch_config3 = 0x0000;
336
337 const ch_setting_t refh = { REF_V, ch_config1, ch_config2, ch_config3 };
338 const ch_setting_t refg = { REF_GND, ch_config1, ch_config2, ch_config3 };
339 const ch_setting_t refc = { REF_COM, ch_config1, ch_config2 & ~0x0080, ch_config3 }; //CH_CHOP:off
340
341 // forcing to set unity-gain and zero-offset
342
343 constexpr raw_t default_gain_coeff_value = 0x1 << 22;
344 constexpr raw_t default_offset_coeff_value = 0;
345
346 reg( GAIN_COEFF0 + gain_index, default_gain_coeff_value );
347 reg( OFFSET_COEFF0 + gain_index, default_offset_coeff_value );
348
349 // measure the logical channel with those different 3 settings
350
351 logical_ch_config( channel_selection, refh );
352 raw_t data_REF = read<raw_t>( channel_selection );
353
354 logical_ch_config( channel_selection, refg );
355 raw_t data_GND = read<raw_t>( channel_selection );
356
357 logical_ch_config( channel_selection, refc );
358 raw_t data_COM = read<raw_t>( channel_selection );
359
360 // calculation
361
362 const double fullscale_voltage = 5.00 / pga_gain[ gain_index ];
363 const double calibrated_gain = (double)(0x1 << 23) * (reference_source_voltage / fullscale_voltage) / (double)(data_REF - data_GND);
364
365#if 1
366 printf( "data_REF = %8ld (%lfV)\r\n", data_REF, raw2v( data_REF, channel_selection ) );
367 printf( "data_GND = %8ld (%lfmV)\r\n", data_GND, raw2mv( data_GND, channel_selection ) );
368 printf( "data_COM = %8ld (%lfmV)\r\n", data_COM, raw2mv( data_COM, channel_selection ) );
369 printf( "gain adjustment = %8lf (%lfdB)\r\n\r\n", calibrated_gain, 20 * log10( calibrated_gain ) );
370#endif
371
372 if ( !( (0.95 < calibrated_gain) && (calibrated_gain < 1.05) ) )
373 return CalibrationError::GainError;
374
375 const double offset_mv = raw2mv( data_COM, channel_selection );
376
377 if ( !( (-10.0 < offset_mv) && (offset_mv < 10.0) ) )
378 return CalibrationError::OffsetError;
379
380 // setting registers: GAIN_COEFF[n] and OFFSET_COEFF[n]
381
382 reg( GAIN_COEFF0 + gain_index, (uint32_t)(default_gain_coeff_value * calibrated_gain) );
383 reg( OFFSET_COEFF0 + gain_index, default_offset_coeff_value + data_COM );
384
385 // if the channel was in-use, revert the setting
386
387 if ( channel_in_use )
388 logical_ch_config( channel_selection, tmp_ch_config );
389 else
390 logical_ch_disable( channel_selection );
391
392 return CalibrationError::NoError;
393}
