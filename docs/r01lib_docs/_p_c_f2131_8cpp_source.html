70 bf[ 5 ] = dec2bcd( cnv_tmp->tm_wday);

73 intfp->bit_op8( SR_Reset, (uint8_t)(~0x80), 0x80 );

151uint8_t PCF2131::int_clear( uint8_t* rv )
152{
153 intfp->reg_r( Control_2, rv, 3 );
154
155 if ( rv[ 0 ] & 0x90 ) // if interrupt flag set in Control_2
156 intfp->reg_w( Control_2, rv[ 0 ] & ~((rv[ 0 ] & 0x90) | 0x49) ); // datasheet 7.11.5
157
158 if ( rv[ 1 ] & 0x08 ) // if interrupt flag set in Control_3
159 intfp->reg_w( Control_3, rv[ 1 ] & ~(0x08) );
160
161 if ( rv[ 2 ] & 0xF0 ) // if interrupt flag set in Control_4
162 intfp->reg_w( Control_4, rv[ 2 ] & ~(rv[ 2 ] & 0xF0) );
163
164 return 0; // dummy
165}

167void PCF2131::periodic_interrupt_enable( periodic_int_select sel, int int_sel )
168{
169 if ( !sel ) {
170 intfp->bit_op8( Control_1, ~0x03, 0x00 );
171 intfp->bit_op8( int_mask_reg[ int_sel ][ 0 ], ~0x30, 0x30 );
172 return;
173 }
174
175 uint8_t v = (sel == EVERY_MINUTE) ? 0x02 : 0x01;
176
177 v=3;
178 intfp->bit_op8( Control_1, ~0x03, v );
179 intfp->bit_op8( int_mask_reg[ int_sel ][ 0 ], ~0x30, ~(v << 4) );
180}
