35void irq_handler( int num )
36{
37 uint32_t flags;
38 flags = GPIO_GpioGetInterruptFlags( gpio_ptr[ num ] );
39 GPIO_GpioClearInterruptFlags( gpio_ptr[ num ], flags );
40
41 for ( int i = 0; i < GPIO_BITS; i++ )
42 if ( cb_table[ num ][ i ] && (flags & (1 << i)) )
43 (cb_table[ num ][ i ])();
44}
