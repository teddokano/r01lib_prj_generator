20float GradationControl::set_gradation( float max_iref, float time, bool up, bool down, float on, float off, bool hold_on, bool hold_off )
21{
22 int iref;
23 float step_duration;
24 float cycle_time;
25 int cycle_time_i = 0;
26 int multi_fctr = 1;
27 int iref_inc = 1;
28 int on_i = 7;
29 int off_i = 7;
30 float ramp_time;
31 uint8_t reg[ 4 ];
32
33 iref = (int)(max_iref * 255.0);
34 time *= 1000.0;
35
36 if ( 0.0 < time ) {
37 step_duration = time / iref;
38
39 if ( 32 < step_duration ) {
40 cycle_time = 8.0;
41 cycle_time_i = 1;
42 }
43 else {
44 cycle_time = 0.5;
45 cycle_time_i = 0;
46 }
47
48 multi_fctr = (int)(step_duration / cycle_time);
49 multi_fctr = (multi_fctr < 1) ? 1 : multi_fctr;
50 multi_fctr = (64 < multi_fctr) ? 64 : multi_fctr;
51
52 if ( multi_fctr == 1 )
53 iref_inc = (int)( iref / (time / cycle_time) );
54 }
55 else {
56 cycle_time = 0;
57 }
58
59 for ( int i = 0; i < 7; i++ ) {
60 if ( on < HOLDTIME[ i + 1 ] ) {
61 on = HOLDTIME[ i ];
62 on_i = i;
63 break;
64 }
65 }
66
67 for ( int i = 0; i < 7; i++ ) {
68 if ( off < HOLDTIME[ i + 1 ] ) {
69 off = HOLDTIME[ i ];
70 off_i = i;
71 break;
72 }
73 }
74
75 reg[ 0 ] = (up << 7) | (down << 6) | (iref_inc - 1); // for RAMP_RATE_GRPn
76 reg[ 1 ] = ( cycle_time_i << 6 ) | (multi_fctr - 1); // for STEP_TIME_GRPn
77 reg[ 2 ] = (hold_on << 7) | (hold_off << 6) | ( on_i << 3) | off_i; // for HOLD_CNTL_GRPn
78 reg[ 3 ] = iref; // for IREF_GRPn
79
80 devp->reg_access( devp->arp[ SETTING ] + group * 4, reg, sizeof( reg ) );
81
82 ramp_time = ((multi_fctr * cycle_time) * (iref / iref_inc)) / 1000;
83
84 cycle_time = on + off;
85 cycle_time += up ? ramp_time : 0;
86 cycle_time += down ? ramp_time : 0;
87
88 cycle = cycle_time;
89
90 return cycle;
91}

107void GradationControl::gradation_group_assign( uint32_t bitmap )
108{
109 const int grp_shift = (devp->arp[ N_GROUP ] == 4) ? 2 : 4;
110 const int n_reg = devp->n_channel / (8 / grp_shift);
111 uint8_t bits = (grp_shift == 2) ? 0x3 : 0x0F;
112 uint8_t reg[ n_reg ];
113 int r_index;
114 int s_index;
115
116 devp->reg_access_r( devp->arp[ GROUP ], reg, sizeof( reg ) );
117
118 for ( int i = 0; i < devp->n_channel; i++ ) {
119 if ( (bitmap >> i) & 0x1 ) {
120 r_index = i / (8 / grp_shift);
121 s_index = (i * grp_shift) % 8;
122
123 reg[ r_index ] = ~(bits << s_index) & reg[ r_index ];

125 }
126 }

131void GradationControl::add_channel( uint8_t ch )
132{

134 uint8_t bits = (grp_shift == 2) ? 0x3 : 0x0F;
135 uint8_t v;

138
139
