44int I2C_device::reg_w( uint8_t reg_adr, const uint8_t *data, uint16_t size )
45{
46 uint8_t buffer[ size + 1 ];
47
48 buffer[ 0 ] = reg_adr;
49 for ( uint16_t i = 0; i < size; i++)
50 buffer[ i + 1 ] = data[ i ];

52 return tx( buffer, sizeof( buffer ) );
