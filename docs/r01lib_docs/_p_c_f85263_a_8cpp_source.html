21void PCF85263A::set( struct tm* now_tmp )
22{
23 // refer datasheet 7.2.6
24
25 time_t now_time;
26 struct tm* cnv_tmp;
27 uint8_t bf[ 10 ];
28
29 bf[ 0 ] = 0x01;
30 bf[ 1 ] = 0xA4;
31 bf[ 2 ] = 0;
32 bf[ 3 ] = dec2bcd( now_tmp->tm_sec );
33 bf[ 4 ] = dec2bcd( now_tmp->tm_min );
34 bf[ 5 ] = dec2bcd( now_tmp->tm_hour );
35 bf[ 6 ] = dec2bcd( now_tmp->tm_mday );
36 bf[ 8 ] = dec2bcd( now_tmp->tm_mon + 1 );
37 bf[ 9 ] = dec2bcd( now_tmp->tm_year - 100 );
38
39 now_time = mktime( now_tmp );
40 cnv_tmp = localtime( &now_time );
41 bf[ 5 ] = dec2bcd( cnv_tmp->tm_wday );
42
43 reg_w( Stop_enable, bf, sizeof( bf ) );
44 reg_w( Stop_enable, 0x00 );
45}

72void PCF85263A::set_alarm( int digit, int val, int int_sel )
73{
74 uint8_t en;
75
76 en = ((val & 80) ? 0x0 : 0x1) << digit;
77
78 reg_w( Second_alarm1 + digit, dec2bcd( val ) );
79 bit_op8( Alarm_enables, ~en, en );
80
81 uint8_t en_bit = (digit < 5) ? 0x10 : 0x80;
82 bit_op8( INTA_enable + int_sel, ~en_bit, en_bit );
83}
