156 if ( (mosi == MB_MOSI) && (miso == MB_MISO) && (sclk == MB_SCK) && (cs == MB_CS) )

170 else

172 panic( "FRDM-MCXA156 SPI on Arduino pin and MikroBus are supported. To use Arduino pins, change jumper setting (short 2-3 pins on R59 and R60) and use \"ARD_MOSI\" and \"ARD_CS\" keywords instead of D10 and D11." );
173 }
174#else
175 unit_base = EXAMPLE_LPSPI_MASTER_BASEADDR;
176 master_clk_freq = LPSPI_MASTER_CLK_FREQ;
