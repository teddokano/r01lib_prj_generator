190status_t I3C::ccc_broadcast( uint8_t ccc, const uint8_t *dp, uint8_t length, bool first_time )
191{
192 uint8_t bp[ REG_RW_BUFFER_SIZE ];
193 status_t r_code;
194
195 bp[ 0 ] = ccc;
196 memcpy( (uint8_t *)bp + 1, (uint8_t *)dp, length );
197
198 if ( first_time || first_broadcast )
199 {
200 first_broadcast = false;
201
202 frequency( 0, 2000000, 2000000 ); // I2C_freq = default, I3C_OD_freq = 2MHz, I3C_PP_freq = 2MHz
203 r_code = write( BROADCAST_ADDR, bp, length + 1 );
204 frequency(); // revert to default frequency
205 }
206 else
207 {
208 r_code = write( BROADCAST_ADDR, bp, length + 1 );
209 }
210 return r_code;
211}

243void I3C::master_ibi_callback( I3C_Type *base, i3c_master_handle_t *handle, i3c_ibi_type_t ibiType, i3c_ibi_state_t ibiState )
244{
245 g_ibiWonFlag = true;
246 g_ibiAddress = handle->ibiAddress;
247
248 switch ( ibiType )
249 {
250 case kI3C_IbiNormal:
251 if ( ibiState == kI3C_IbiDataBuffNeed )
252 {
253 handle->ibiBuff = g_ibiBuff;
254 }
255 else
256 {
257 memcpy( g_ibiUserBuff, (void *)handle->ibiBuff, handle->ibiPayloadSize );
258 g_ibiUserBuffUsed = handle->ibiPayloadSize;
259 }
260 break;
261
262 default:
263 assert(false);
264 break;
265 }
266
267 if ( g_ibi_callback )
268 g_ibi_callback();
269}
