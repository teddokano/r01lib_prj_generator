96void PCA8561::char2seg( int pos, int c )
97{
98 c = toupper( c );
99
100 if ( (c < 32) || (92 < c) )
101 c = 32;
102
103 uint16_t p = char_pattern[ c - 32 ];
104
105 uint8_t c0 = p & 0x0F;
106 uint8_t c1 = (p >> 4) & 0x0F;
107 uint8_t c2 = (p >> 8) & 0x0F;
108 uint8_t c3 = (p >> 12) & 0x0F;
109
110 bf[ pos / 2 + 0 ] &= ~(0x0F << (4 * (pos % 2)));
111 bf[ pos / 2 + 3 ] &= ~(0x0F << (4 * (pos % 2)));
112 bf[ pos / 2 + 6 ] &= ~(0x0F << (4 * (pos % 2)));
113 bf[ pos / 2 + 9 ] &= ~(0x0F << (4 * (pos % 2)));
114
115 bf[ pos / 2 + 0 ] |= c0 << (4 * (pos % 2));
116 bf[ pos / 2 + 3 ] |= c1 << (4 * (pos % 2));
117 bf[ pos / 2 + 6 ] |= c2 << (4 * (pos % 2));
118 bf[ pos / 2 + 9 ] |= c3 << (4 * (pos % 2));
119}
